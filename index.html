function bubblesortOnce(a) {
  let newA = a.slice();
  for (let i=1; i < newA.length; i++) {
    if (newA[i-1] > newA[i]) {
      [newA[i-1], newA[i]] = [newA[i], newA[i-1]];
    }
  }
  return newA
}

function shiftSlide(dir) {
  items.classList.add('shifting');
  if (allowShift) {
    posInitial = items.offsetTop;

    if (dir == 1) {
      items.style.top = (posInitial - slideSize) + "px";
      index++;
    } else if (dir == -1) {
      items.style.top = (posInitial + slideSize) + "px";
      index--;
    }
  };
  allowShift = false;
}
